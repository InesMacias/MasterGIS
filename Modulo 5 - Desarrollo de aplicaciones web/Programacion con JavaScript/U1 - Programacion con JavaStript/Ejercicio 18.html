Ejercicio 18

Eventos

Haz que cuando se haga click sobre el botón “Guardar”, salté un alert con un mensaje de guardado.



Cuando hagamos foco sobre este input se pondrá de color amarillo y cuando se lo quitemos cambiará de color



Dependiendo si escribamos una vocal o una consonante el contenido del input apare cerá de distinto color

                            

<script>
document.querySelector('.btn').addEventListener('click', function () {
    alert('Guardado');
});
</script>

<script>
const inputNombre = document.getElementById('txtNombre');

inputNombre.addEventListener('focus', function() {
    inputNombre.style.backgroundColor = 'yellow';
});

inputNombre.addEventListener('blur', function() {
    inputNombre.style.backgroundColor = '';
});
</script>

<script>
const inputColor = document.getElementById('txtColor');

inputColor.addEventListener('keydown', function(event) {
    const texto = inputColor.innerText;

    if (texto) { // Verificar si 'texto' no es undefined
        let resultado = '';

        for (let i = 0; i < texto.length; i++) {
            const letra = texto[i].toLowerCase();

            if (letra.match(/[aeiou]/i)) {
                resultado += `<span style="color: red;">${letra}</span>`;
            } else if (letra.match(/[a-z]/i)) {
                resultado += `<span style="color: blue;">${letra}</span>`;
            } else {
                resultado += letra;
            }
        }

        inputColor.innerHTML = resultado;

        // pone el cursor al final de
        const selection = window.getSelection();
        const range = document.createRange();
        range.selectNodeContents(inputColor);
        range.collapse(false);
        selection.removeAllRanges();
        selection.addRange(range);
    }
});
</script>
